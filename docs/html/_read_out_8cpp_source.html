33G4double ReadOut::closest_value(std::vector<G4double> vec, G4double value) {
34 // Set closest do the maximum value of G4double
35 G4double closest = std::numeric_limits<double>::max();
36 // sort the vector
37 std::sort(vec.begin(), vec.end());
38
39 // iterate on the vector and compute the distance between the value and the element of the vector
40 // When the distance starts to grow stop the loop
41 for (auto elem : vec) {
42 if (std::abs(elem - value) < std::abs(closest - value)) {
43 closest = elem;
44 } else {

46 }
