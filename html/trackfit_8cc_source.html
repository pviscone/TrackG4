43FitData Fit(Event *event) {
44 FitData fitdata;

46 TF1 line("line", "pol1", -GeometryParameters::worldSize, GeometryParameters::worldSize);
47
48 // Store the hits
49 std::vector<double> x = (event->detectorData).posX;
50 std::vector<double> y = (event->detectorData).posY;
51 std::vector<double> z = (event->detectorData).posZ;
52 std::vector<double> dx, dy;
53 std::vector<double> dz(x.size(), GeometryParameters::moduleDimZ / 2);
54 // Create vectors of errors. The even layers are pixel module, the odds are strips.
55 for (auto lay : (event->detectorData).Layer) {
56 if (lay % 2 == 0) {
57 dx.push_back(ReadOutParameters::pixelDimX / 2);
58 dy.push_back(ReadOutParameters::pixelDimY / 2);
59 } else {
60 dx.push_back(ReadOutParameters::stripDimX / 2);
61 dy.push_back(ReadOutParameters::stripDimY / 2);
62 }
63 }
64 // Create the graph containing the hits in the ZX plane projection and fill the struct
65 TGraphErrors GR_zx(x.size(), &z[0], &x[0], &dz[0], &dx[0]);
66 GR_zx.Fit("line");
67 fitdata.x0 = line.GetParameter(0);
68 fitdata.x0_err = line.GetParError(0);
69 fitdata.mx = line.GetParameter(1);
70 fitdata.mx_err = line.GetParError(1);
71 fitdata.chi2zx = line.GetChisquare();
72 // Create the graph containing the hits in the ZY plane projection and fill the struct
73 TGraphErrors GR_zy(y.size(), &z[0], &y[0], &dz[0], &dy[0]);
74 GR_zy.Fit("line");
75 fitdata.y0 = line.GetParameter(0);
76 fitdata.y0_err = line.GetParError(0);
77 fitdata.my = line.GetParameter(1);
78 fitdata.my_err = line.GetParError(1);
