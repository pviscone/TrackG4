53FitData Fit(Event *event, int i, bool saveFigs = false) {
54 FitData fitdata;
55 std::string name, title;
56 double prev_x0, prev_y0, prev_mx, prev_my;
57 // create the linear function to fit (a line in the range of the world size)
58 TF1 line("line", "pol1", -GeometryParameters::worldSize, GeometryParameters::worldSize);
59
60 // Store the hits
61 std::vector<double> x = (event->detectorData).posX;
62 std::vector<double> y = (event->detectorData).posY;
63 std::vector<double> z = (event->detectorData).posZ;
64 std::vector<double> dx, dy;
65 std::vector<double> dz(x.size(), GeometryParameters::moduleDimZ / sqrt(12));
66 // Create vectors of errors. The even layers are pixel module, the odds are strips.
67 for (auto lay : (event->detectorData).Layer) {
68 if (lay % 2 == 0) {
69 dx.push_back(ReadOutParameters::pixelDimX / sqrt(12));
70 dy.push_back(ReadOutParameters::pixelDimY / sqrt(12));
71 } else {
72 dx.push_back(ReadOutParameters::stripDimX / sqrt(12));
73 dy.push_back(ReadOutParameters::stripDimY / sqrt(12));
74 }
75 }
76 // Create the graph containing the hits in the ZX plane projection and fill the struct
77 TCanvas c1("c1", "c1", 800, 600);
78 TGraphErrors GR_zx(x.size(), &z[0], &x[0], &dz[0], &dx[0]);
79 prev_mx = (x.back() - x[0]) / (z.back() - z[0]);
80 prev_x0 = x[0] - prev_mx * z[0];
81 line.SetParameter(0, prev_x0);
82 line.SetParameter(1, prev_mx);
83 auto status_zx = GR_zx.Fit("line", "Q");
84 fitdata.x0 = line.GetParameter(0);
85 fitdata.x0_err = line.GetParError(0);
86 fitdata.mx = line.GetParameter(1);
87 fitdata.mx_err = line.GetParError(1);
88 fitdata.chi2zx = line.GetChisquare();
89 title = "ZX evID:" + std::to_string(i);
90 GR_zx.SetTitle(title.c_str());
91 GR_zx.GetXaxis()->SetTitle("Z [mm]");
92 GR_zx.GetYaxis()->SetTitle("X [mm]");
93 GR_zx.Draw("AP");
94 if (saveFigs) {
95 name = "../data/fit_img/zx_" + std::to_string(i) + ".png";
96 c1.SaveAs(name.c_str());
97 }
98 // Create the graph containing the hits in the ZY plane projection and fill the struct
99 TCanvas c2("c2", "c2", 800, 600);
100 TGraphErrors GR_zy(y.size(), &z[0], &y[0], &dz[0], &dy[0]);
101 prev_my = (y.back() - y[0]) / (z.back() - z[0]);
102 prev_y0 = y[0] - prev_my * z[0];
103 line.SetParameter(0, prev_y0);
104 line.SetParameter(1, prev_my);
105 auto status_zy = GR_zy.Fit("line", "Q");
106 fitdata.y0 = line.GetParameter(0);
107 fitdata.y0_err = line.GetParError(0);
108 fitdata.my = line.GetParameter(1);
109 fitdata.my_err = line.GetParError(1);
110 fitdata.chi2zy = line.GetChisquare();
111 title = "ZY evID:" + std::to_string(i);
112 GR_zy.SetTitle(title.c_str());
113 GR_zy.GetXaxis()->SetTitle("Z [mm]");
114 GR_zy.GetYaxis()->SetTitle("Y [mm]");
115 GR_zy.Draw("AP");
116 if (saveFigs) {
117 name = "../data/fit_img/zy_" + std::to_string(i) + ".png";
118 c2.SaveAs(name.c_str());
119 }
120 // Check that the fit on both projection was successful
121 fitdata.status = status_zx && status_zy;
122 return fitdata;
123}
